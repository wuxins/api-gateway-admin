public class org.springframework.http.HttpStatus implements Serializable{

    private int value; //
    private enum series; //
    private String reasonPhrase; //

    public int getValue(){
        return value;
    }

    public void setValue(int value){
        this.value = value;
    }

    public enum getSeries(){
        return series;
    }

    public void setSeries(enum series){
        this.series = series;
    }

    public String getReasonPhrase(){
        return reasonPhrase;
    }

    public void setReasonPhrase(String reasonPhrase){
        this.reasonPhrase = reasonPhrase;
    }

}
